Podstawy skryptów Bash

Struktura skryptu

Każdy skrypt bash powinien zaczynać się od shebang:

#!/bin/bash

Shebang (#!) informuje system operacyjny, który interpreter ma być użyty do wykonania skryptu. /bin/bash to ścieżka do interpretera Bash.

Nadawanie uprawnień wykonywania

chmod +x skrypt.sh    # Nadanie uprawnień
./skrypt.sh           # Uruchomienie skryptu

Zmienne

# Deklaracja zmiennej (bez spacji wokół =)
nazwa="Jan"
wiek=25

# Odwołanie do zmiennej
echo $nazwa
echo ${nazwa}    # Preferowana składnia

# Zmienne środowiskowe
echo $HOME       # Katalog domowy użytkownika
echo $USER       # Nazwa użytkownika
echo $PATH       # Ścieżka wyszukiwania programów

Argumenty skryptu

$0    # Nazwa skryptu
$1    # Pierwszy argument
$2    # Drugi argument
$#    # Liczba argumentów
$@    # Wszystkie argumenty jako oddzielne słowa
$*    # Wszystkie argumenty jako jeden string

Przykład:

#!/bin/bash
echo "Nazwa skryptu: $0"
echo "Pierwszy argument: $1"
echo "Liczba argumentów: $#"
echo "Wszystkie argumenty: $@"

Instrukcje warunkowe

If-elif-else

if [ warunek ]; then
    # kod
elif [ inny_warunek ]; then
    # kod
else
    # kod
fi

Operatory porównania dla liczb

-eq    # równy (equal)
-ne    # nierówny (not equal)
-lt    # mniejszy (less than)
-le    # mniejszy lub równy (less or equal)
-gt    # większy (greater than)
-ge    # większy lub równy (greater or equal)

Przykład:

if [ $wiek -ge 18 ]; then
    echo "Pełnoletni"
else
    echo "Niepełnoletni"
fi

Operatory dla stringów

=      # równy
!=     # nierówny
-z     # pusty string
-n     # niepusty string

Przykład:

if [ "$nazwa" = "Jan" ]; then
    echo "Witaj Jan!"
fi

if [ -z "$zmienna" ]; then
    echo "Zmienna jest pusta"
fi

Operatory dla plików

-f     # plik istnieje i jest zwykłym plikiem
-d     # katalog istnieje
-e     # plik lub katalog istnieje
-r     # plik ma uprawnienia do odczytu
-w     # plik ma uprawnienia do zapisu
-x     # plik ma uprawnienia do wykonania
-s     # plik istnieje i nie jest pusty

Przykład:

if [ -f "plik.txt" ]; then
    echo "Plik istnieje"
fi

if [ ! -d "katalog" ]; then
    echo "Katalog nie istnieje"
    mkdir katalog
fi

Podwójne nawiasy kwadratowe [[]]

Nowsza składnia z dodatkowymi możliwościami:

if [[ $zmienna =~ ^[0-9]+$ ]]; then
    echo "To liczba"
fi

if [[ "$string" == *"substring"* ]]; then
    echo "Zawiera substring"
fi

Pętle

Pętla for

# Iteracja po liczbach
for i in {1..10}; do
    echo $i
done

# Iteracja po literach
for letter in {a..z}; do
    echo $letter
done

# Iteracja po plikach
for file in *.txt; do
    echo "Przetwarzam: $file"
done

# Składnia C-style
for (( i=0; i<10; i++ )); do
    echo $i
done

Pętla while

counter=0
while [ $counter -lt 10 ]; do
    echo $counter
    counter=$((counter + 1))
done

# Czytanie pliku linia po linii
while IFS= read -r line; do
    echo "$line"
done < plik.txt

Pętla until

counter=0
until [ $counter -ge 10 ]; do
    echo $counter
    counter=$((counter + 1))
done

Funkcje

# Definicja funkcji
nazwa_funkcji() {
    local zmienna_lokalna="wartość"
    echo "Argument: $1"
    return 0
}

# Wywołanie funkcji
nazwa_funkcji "argument"
wynik=$?    # Kod powrotu (return value)

Operacje arytmetyczne

# Podstawowe operacje
wynik=$((5 + 3))        # 8
wynik=$((10 - 4))       # 6
wynik=$((6 * 7))        # 42
wynik=$((20 / 4))       # 5
wynik=$((17 % 5))       # 2 (reszta z dzielenia)

# Inkrementacja
((counter++))
((counter--))
counter=$((counter + 1))

# Operacje na zmiennych
a=10
b=5
suma=$((a + b))

Tablice

# Deklaracja tablicy
owoce=("jabłko" "banan" "gruszka")

# Dostęp do elementów
echo ${owoce[0]}        # jabłko
echo ${owoce[1]}        # banan

# Wszystkie elementy
echo ${owoce[@]}

# Liczba elementów
echo ${#owoce[@]}

# Dodawanie elementów
owoce+=("pomarańcza")

# Iteracja po tablicy
for owoc in "${owoce[@]}"; do
    echo $owoc
done

Struktury katalogów i plików

Tworzenie katalogów

Podstawowe komendy

mkdir katalog           # Tworzy katalog
mkdir -p a/b/c         # Tworzy katalogi rekurencyjnie
rmdir katalog          # Usuwa pusty katalog
rm -r katalog          # Usuwa katalog z zawartością

Tworzenie wielu katalogów

# Użycie ekspansji nawiasów klamrowych
mkdir {dir1,dir2,dir3}
mkdir folder{1..10}

# Kombinacje
mkdir {a,b,c}{1,2,3}
# Tworzy: a1, a2, a3, b1, b2, b3, c1, c2, c3

Tworzenie plików

touch plik.txt                    # Tworzy pusty plik
echo "treść" > plik.txt          # Tworzy plik z treścią (nadpisuje)
echo "treść" >> plik.txt         # Dodaje treść do pliku
cat > plik.txt << EOF            # Heredoc
Linia 1
Linia 2
EOF

Nawigacja

cd katalog              # Przejście do katalogu
cd ..                   # Katalog wyżej
cd ~                    # Katalog domowy
cd -                    # Poprzedni katalog
pwd                     # Wyświetl obecną ścieżkę

Projekt: Szachownica

Analiza problemu: - Szachownica ma 8 kolumn (a-h) i 8 wierszy (1-8) - Każde pole ma kolor: białe lub czarne - Kolor zależy od sumy współrzędnych (parzysta suma = białe, nieparzysta = czarne) - Niektóre pola mają figury startowe

Szczegółowe rozwiązanie:

#!/bin/bash

# Tworzenie głównego katalogu
mkdir -p szachownica
cd szachownica || exit

# Funkcja określająca kolor pola
# Algorytm: jeśli suma numeru kolumny i wiersza jest parzysta -> białe
# W przeciwnym razie -> czarne
get_color() {
    local col=$1
    local row=$2
    
    # Konwersja litery na liczbę (a=1, b=2, ..., h=8)
    local col_num=$(printf "%d" "'$col")
    col_num=$((col_num - 96))
    
    # Sprawdzenie parzystości sumy
    if [ $(( (col_num + row) % 2 )) -eq 0 ]; then
        echo "bialy"
    else
        echo "czarny"
    fi
}

# Funkcja zwracająca symbol figury na danym polu
get_piece() {
    local col=$1
    local row=$2
    
    # Białe piony (wiersz 2)
    if [ $row -eq 2 ]; then
        echo "WP"
        return
    fi
    
    # Białe figury (wiersz 1)
    if [ $row -eq 1 ]; then
        case $col in
            a|h) echo "WR" ;;  # Wieża (Rook)
            b|g) echo "WN" ;;  # Skoczek (kNight)
            c|f) echo "WB" ;;  # Goniec (Bishop)
            d) echo "WQ" ;;    # Hetman (Queen)
            e) echo "WK" ;;    # Król (King)
        esac
        return
    fi
    
    # Czarne piony (wiersz 7)
    if [ $row -eq 7 ]; then
        echo "BP"
        return
    fi
    
    # Czarne figury (wiersz 8)
    if [ $row -eq 8 ]; then
        case $col in
            a|h) echo "BR" ;;
            b|g) echo "BN" ;;
            c|f) echo "BB" ;;
            d) echo "BQ" ;;
            e) echo "BK" ;;
        esac
        return
    fi
    
    # Puste pole
    echo ""
}

# Główna pętla generująca strukturę
for col in {a..h}; do
    for row in {1..8}; do
        # Nazwa pola (np. a1, e4, h8)
        field="${col}${row}"
        
        # Tworzenie katalogu dla pola
        mkdir -p "$field"
        
        # Określenie koloru pola
        color=$(get_color $col $row)
        
        # Określenie figury (jeśli jest)
        piece=$(get_piece $col $row)
        
        # Tworzenie pliku z kolorem i ewentualnie figurą
        if [ -n "$piece" ]; then
            echo "$piece" > "$field/$color"
        else
            touch "$field/$color"
        fi
        
        echo "Utworzono: $field/$color $([ -n "$piece" ] && echo "[$piece]")"
    done
done

echo ""
echo "Szachownica została utworzona!"
echo "Struktura: szachownica/[a-h][1-8]/[bialy|czarny]"

Szczegóły implementacji:

  1. Konwersja literki na liczbę:

    local col_num=$(printf "%d" "'$col")
col_num=$((col_num - 96))
    • printf "%d" "'$col" - zwraca kod ASCII litery
    • Odejmujemy 96, aby a=1, b=2, itd.

  2. Sprawdzanie parzystości:

    if [ $(( (col_num + row) % 2 )) -eq 0 ]; then
    • Operator % zwraca resztę z dzielenia
    • Jeśli reszta to 0, liczba jest parzysta

  3. Instrukcja case:

    case $col in
    a|h) echo "WR" ;;
    b|g) echo "WN" ;;
esac
    • Pozwala na wielokrotne dopasowanie wzorców
    • | oznacza “lub”
    • ;; kończy każdy przypadek

Operacje na plikach tekstowych

Czytanie plików

# Wyświetlenie całego pliku
cat plik.txt

# Pierwsze 10 linii
head plik.txt
head -n 5 plik.txt    # Pierwsze 5 linii

# Ostatnie 10 linii
tail plik.txt
tail -n 5 plik.txt    # Ostatnie 5 linii

# Czytanie linia po linii
while IFS= read -r line; do
    echo "$line"
done < plik.txt

Zmienne IFS (Internal Field Separator)

IFS określa separator pól. Domyślnie to spacja, tab i newline.

# Czytanie CSV z separatorem średnika
while IFS=';' read -r col1 col2 col3; do
    echo "Kolumna 1: $col1"
    echo "Kolumna 2: $col2"
    echo "Kolumna 3: $col3"
done < dane.csv

Przykład z zachowaniem oryginalnego IFS:

OLD_IFS=$IFS
IFS=';'
# operacje
IFS=$OLD_IFS

Grep - wyszukiwanie wzorców

# Podstawowe wyszukiwanie
grep "wzorzec" plik.txt

# Ignorowanie wielkości liter
grep -i "wzorzec" plik.txt

# Inwersja (linie NIE zawierające wzorca)
grep -v "wzorzec" plik.txt

# Wyświetlanie numerów linii
grep -n "wzorzec" plik.txt

# Rekurencyjne przeszukiwanie katalogów
grep -r "wzorzec" katalog/

# Zliczanie wystąpień
grep -c "wzorzec" plik.txt

# Kontekst (linie przed i po)
grep -A 2 "wzorzec" plik.txt    # 2 linie po
grep -B 2 "wzorzec" plik.txt    # 2 linie przed
grep -C 2 "wzorzec" plik.txt    # 2 linie przed i po

Sed - edytor strumieniowy

# Podstawianie (pierwsze wystąpienie w linii)
sed 's/stary/nowy/' plik.txt

# Podstawianie (wszystkie wystąpienia)
sed 's/stary/nowy/g' plik.txt

# Modyfikacja pliku w miejscu
sed -i 's/stary/nowy/g' plik.txt

# Usuwanie linii
sed '/wzorzec/d' plik.txt        # Usuwa linie zawierające wzorzec
sed '5d' plik.txt                # Usuwa 5. linię
sed '2,5d' plik.txt              # Usuwa linie 2-5

# Drukowanie tylko dopasowanych linii
sed -n '/wzorzec/p' plik.txt

Awk - przetwarzanie tekstu

# Wyświetlenie konkretnych kolumn
awk '{print $1}' plik.txt        # Pierwsza kolumna
awk '{print $1, $3}' plik.txt    # Pierwsza i trzecia

# Zmiana separatora
awk -F';' '{print $1}' dane.csv

# Warunkowe przetwarzanie
awk '$3 > 100 {print $1}' plik.txt

# Obliczenia
awk '{suma += $1} END {print suma}' plik.txt

Projekt: Usuwanie linii z pliku

Analiza problemu: - Skrypt przyjmuje 2 argumenty: nazwę pliku i frazę - Usuwa wszystkie linie zawierające podaną frazę - Powinien tworzyć kopię zapasową

Szczegółowe rozwiązanie:

#!/bin/bash

# Walidacja liczby argumentów
if [ $# -ne 2 ]; then
    echo "Błąd: Nieprawidłowa liczba argumentów"
    echo "Użycie: $0 <nazwa_pliku> <fraza_do_usunięcia>"
    echo ""
    echo "Przykład: $0 dane.txt 'error'"
    exit 1
fi

# Przypisanie argumentów do zmiennych (dla czytelności)
file=$1
phrase=$2

# Sprawdzenie czy plik istnieje
if [ ! -f "$file" ]; then
    echo "Błąd: Plik '$file' nie istnieje!"
    exit 1
fi

# Sprawdzenie uprawnień do odczytu
if [ ! -r "$file" ]; then
    echo "Błąd: Brak uprawnień do odczytu pliku '$file'"
    exit 1
fi

# Sprawdzenie uprawnień do zapisu
if [ ! -w "$file" ]; then
    echo "Błąd: Brak uprawnień do zapisu pliku '$file'"
    exit 1
fi

# Tworzenie kopii zapasowej z timestampem
timestamp=$(date +%Y%m%d_%H%M%S)
backup_file="${file}.backup_${timestamp}"
cp "$file" "$backup_file"

if [ $? -eq 0 ]; then
    echo "Utworzono kopię zapasową: $backup_file"
else
    echo "Błąd: Nie udało się utworzyć kopii zapasowej!"
    exit 1
fi

# Zliczanie linii przed operacją
lines_before=$(wc -l < "$file")

# Usunięcie linii zawierających frazę
# grep -v wybiera linie NIE zawierające wzorca
grep -v "$phrase" "$file" > "${file}.tmp"

# Sprawdzenie czy operacja się powiodła
if [ $? -eq 0 ]; then
    mv "${file}.tmp" "$file"
    
    # Zliczanie linii po operacji
    lines_after=$(wc -l < "$file")
    lines_removed=$((lines_before - lines_after))
    
    echo "Operacja zakończona sukcesem!"
    echo "Usunięto $lines_removed linii zawierających frazę: '$phrase'"
    echo "Linie przed: $lines_before"
    echo "Linie po: $lines_after"
else
    echo "Błąd podczas przetwarzania pliku!"
    rm -f "${file}.tmp"
    exit 1
fi

Szczegóły implementacji:

  1. Walidacja argumentów:

    if [ $# -ne 2 ]; then
    • $# zawiera liczbę argumentów
    • -ne oznacza “not equal”

  2. Sprawdzanie pliku:

    [ ! -f "$file" ]     # Plik nie istnieje
[ ! -r "$file" ]     # Brak uprawnień do odczytu
[ ! -w "$file" ]     # Brak uprawnień do zapisu

  3. Timestamp w nazwie backup:

    timestamp=$(date +%Y%m%d_%H%M%S)
    • Format: YYYYMMDD_HHMMSS
    • Przykład: 20260127_143052

  4. Sprawdzanie kodu powrotu:

    if [ $? -eq 0 ]; then
    • $? zawiera kod powrotu ostatniej komendy
    • 0 oznacza sukces, inne wartości to błędy

Alternatywne metody:

# Użycie sed
sed -i.backup "/$phrase/d" "$file"

# Użycie awk
awk "!/$phrase/" "$file" > temp && mv temp "$file"

Generowanie liczb losowych

Zmienna RANDOM

Bash ma wbudowaną zmienną $RANDOM, która zwraca losową liczbę z zakresu 0-32767.

# Podstawowe użycie
echo $RANDOM

# Liczba z zakresu 0-9
echo $((RANDOM % 10))

# Liczba z zakresu 1-10
echo $((RANDOM % 10 + 1))

# Liczba z zakresu min-max
min=5
max=15
echo $((RANDOM % (max - min + 1) + min))

Losowanie z tablicy

# Tablica z opcjami
opcje=("kamień" "papier" "nożyce")

# Losowy indeks
indeks=$((RANDOM % ${#opcje[@]}))

# Wybór losowego elementu
wybor=${opcje[$indeks]}
echo $wybor

Projekt: Kamień, Papier, Nożyce

Analiza problemu: - Gracz wybiera K, P lub N - Komputer losuje swoją opcję - Trzeba określić zwycięzcę według reguł: - Kamień > Nożyce - Papier > Kamień - Nożyce > Papier

Szczegółowe rozwiązanie:

#!/bin/bash

# Kolory dla lepszej prezentacji (opcjonalne)
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Tablice z opcjami
choices=("K" "P" "N")
names=("Kamień" "Papier" "Nożyce")
emojis=("🪨" "📄" "✂️")

# Funkcja wyświetlająca zasady
show_rules() {
    echo -e "${BLUE}=== ZASADY GRY ===${NC}"
    echo "Kamień ${emojis[0]} wygrywa z Nożycami ${emojis[2]}"
    echo "Papier ${emojis[1]} wygrywa z Kamieniem ${emojis[0]}"
    echo "Nożyce ${emojis[2]} wygrywają z Papierem ${emojis[1]}"
    echo ""
}

# Funkcja określająca zwycięzcę
# Zwraca: 0 = remis, 1 = gracz wygrywa, 2 = komputer wygrywa
determine_winner() {
    local player=$1
    local computer=$2
    
    if [ "$player" == "$computer" ]; then
        return 0  # Remis
    fi
    
    # Przypadki wygranej gracza
    if [ "$player" == "K" ] && [ "$computer" == "N" ]; then
        return 1
    elif [ "$player" == "P" ] && [ "$computer" == "K" ]; then
        return 1
    elif [ "$player" == "N" ] && [ "$computer" == "P" ]; then
        return 1
    else
        return 2  # Komputer wygrywa
    fi
}

# Funkcja konwertująca symbol na indeks
get_index() {
    local choice=$1
    case $choice in
        K) echo 0 ;;
        P) echo 1 ;;
        N) echo 2 ;;
    esac
}

# Wyświetlenie nagłówka
echo -e "${BLUE}"
echo "╔═══════════════════════════════════════╗"
echo "║   KAMIEŃ, PAPIER, NOŻYCE - GRA        ║"
echo "╚═══════════════════════════════════════╝"
echo -e "${NC}"

show_rules

# Licznik punktów (dla wielu rund)
player_score=0
computer_score=0
draws=0

# Pętla gry
while true; do
    echo -e "${YELLOW}Wybierz:${NC}"
    echo "  K - Kamień ${emojis[0]}"
    echo "  P - Papier ${emojis[1]}"
    echo "  N - Nożyce ${emojis[2]}"
    echo "  Q - Wyjście"
    echo ""
    
    read -p "Twój wybór: " player
    
    # Konwersja na wielkie litery
    player=$(echo "$player" | tr '[:lower:]' '[:upper:]')
    
    # Sprawdzenie wyjścia
    if [ "$player" == "Q" ]; then
        echo ""
        echo -e "${BLUE}=== KOŃCOWE WYNIKI ===${NC}"
        echo "Gracz: $player_score"
        echo "Komputer: $computer_score"
        echo "Remisy: $draws"
        echo ""
        echo "Dziękujemy za grę!"
        exit 0
    fi
    
    # Walidacja wyboru
    if [[ ! "$player" =~ ^[KPN]$ ]]; then
        echo -e "${RED}Błędny wybór! Wybierz K, P, N lub Q${NC}"
        echo ""
        continue
    fi
    
    # Losowanie wyboru komputera
    computer_index=$((RANDOM % 3))
    computer=${choices[$computer_index]}
    
    # Wyświetlenie wyborów
    echo ""

Systemy liczbowe

Podstawy systemów liczbowych

System dziesiętny (decimal): - Podstawa: 10 - Cyfry: 0-9 - Przykład: 123 = 1×10² + 2×10¹ + 3×10⁰

System binarny (binary): - Podstawa: 2 - Cyfry: 0-1 - Przykład: 1011 = 1×2³ + 0×2² + 1×2¹ + 1×2⁰ = 8 + 0 + 2 + 1 = 11

System ósemkowy (octal): - Podstawa: 8 - Cyfry: 0-7 - Przykład: 17 = 1×8¹ + 7×8⁰ = 8 + 7 = 15

System szesnastkowy (hexadecimal): - Podstawa: 16 - Cyfry: 0-9, A-F - Przykład: 1F = 1×16¹ + 15×16⁰ = 16 + 15 = 31

Konwersje w Bash

# Binarny na dziesiętny
echo $((2#1011))        # Wynik: 11

# Ósemkowy na dziesiętny
echo $((8#17))          # Wynik: 15

# Szesnastkowy na dziesiętny
echo $((16#1F))         # Wynik: 31

# Dziesiętny na binarny (używając bc)
echo "obase=2; 11" | bc  # Wynik: 1011

# Dziesiętny na hex
printf "%x\n" 31         # Wynik: 1f
printf "%X\n" 31         # Wynik: 1F (wielkie litery)

Projekt: Konwerter binarny → dziesiętny

Analiza problemu: - Sprawdzić czy liczba jest binarna (tylko 0 i 1) - Przekonwertować na system dziesiętny - Obsłużyć błędne dane

Szczegółowe rozwiązanie:

#!/bin/bash

# Kolory
RED='\033[0;31m'
GREEN='\033[0;32m'
BLUE='\033[0;34m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Funkcja sprawdzająca czy string jest liczbą binarną
is_binary() {
    local num=$1
    
    # Sprawdzenie czy string jest pusty
    if [ -z "$num" ]; then
        return 1
    fi
    
    # Wyrażenie regularne: ^ = początek, $ = koniec, + = jeden lub więcej
    if [[ $num =~ ^[01]+$ ]]; then
        return 0  # Jest binarny
    else
        return 1  # Nie jest binarny
    fi
}

# Funkcja konwertująca binarny na dziesiętny
binary_to_decimal() {
    local binary=$1
    
    # Metoda 1: Użycie Bash
    echo $((2#$binary))
    
    # Metoda 2: Ręczne obliczenie (dla edukacji)
    # local decimal=0
    # local power=0
    # local length=${#binary}
    # for (( i=length-1; i>=0; i-- )); do
    #     digit=${binary:$i:1}
    #     decimal=$((decimal + digit * (2 ** power)))
    #     ((power++))
    # done
    # echo $decimal
}

# Funkcja konwertująca dziesiętny na binarny (bonus)
decimal_to_binary() {
    local decimal=$1
    local binary=""
    
    if [ $decimal -eq 0 ]; then
        echo "0"
        return
    fi
    
    while [ $decimal -gt 0 ]; do
        remainder=$((decimal % 2))
        binary="${remainder}${binary}"
        decimal=$((decimal / 2))
    done
    
    echo $binary
}

# Funkcja wyświetlająca proces konwersji
show_conversion_steps() {
    local binary=$1
    local length=${#binary}
    local decimal=0
    
    echo -e "${YELLOW}Proces konwersji:${NC}"
    echo -n "$binary = "
    
    for (( i=0; i<length; i++ )); do
        digit=${binary:$i:1}
        power=$((length - i - 1))
        value=$((digit * (2 ** power)))
        decimal=$((decimal + value))
        
        if [ $i -gt 0 ]; then
            echo -n " + "
        fi
        echo -n "${digit}×2^${power}"
    done
    
    echo " = $decimal"
    echo ""
}

# Nagłówek
echo -e "${BLUE}"
echo "╔═══════════════════════════════════════════╗"
echo "║  KONWERTER: BINARNY ⟷ DZIESIĘTNY         ║"
echo "╚═══════════════════════════════════════════╝"
echo -e "${NC}"

# Sprawdzenie czy podano argumenty
if [ $# -gt 0 ]; then
    # Tryb z argumentami
    echo -e "${BLUE}Tryb wsadowy:${NC}"
    echo ""
    
    for binary in "$@"; do
        if is_binary "$binary"; then
            decimal=$(binary_to_decimal "$binary")
            echo -e "${GREEN}✓${NC} $binary (bin) = ${GREEN}$decimal${NC} (dec)"
            
            # Opcjonalnie: pokaż kroki
            if [ ${#binary} -le 8 ]; then
                show_conversion_steps "$binary"
            fi
        else
            echo -e "${RED}✗ Błąd:${NC} '$binary' nie jest liczbą binarną!"
            echo "  Liczba binarna może zawierać tylko cyfry 0 i 1"
            echo ""
        fi
    done
else
    # Tryb interaktywny
    echo "Tryb interaktywny - wpisz liczbę binarną lub komendę:"
    echo "  - Liczba binarna (np. 1011) - konwersja do dziesiętnego"
    echo "  - d:liczba (np. d:15) - konwersja dziesiętnego na binarny"
    echo "  - h - pomoc"
    echo "  - q - wyjście"
    echo ""
    
    while true; do
        read -p ">> " input
        
        # Usunięcie białych znaków
        input=$(echo "$input" | tr -d ' ')
        
        # Sprawdzenie komend specjalnych
        case $input in
            q|Q|quit|exit)
                echo "Do widzenia!"
                exit 0
                ;;
            h|H|help|pomoc)
                echo ""
                echo "Pomoc:"
                echo "  1011         - konwertuje 1011 (bin) na dziesiętny"
                echo "  d:15         - konwertuje 15 (dec) na binarny"
                echo "  q            - wyjście"
                echo ""
                continue
                ;;
            d:*|D:*)
                # Konwersja dziesiętny -> binarny
                decimal=${input#*:}
                if [[ $decimal =~ ^[0-9]+$ ]]; then
                    binary=$(decimal_to_binary $decimal)
                    echo -e "${GREEN}Wynik:${NC} $decimal (dec) = ${GREEN}$binary${NC} (bin)"
                else
                    echo -e "${RED}Błąd:${NC} '$decimal' nie jest liczbą dziesiętną!"
                fi
                echo ""
                continue
                ;;
        esac
        
        # Konwersja binarny -> dziesiętny
        if is_binary "$input"; then
            decimal=$(binary_to_decimal "$input")
            echo -e "${GREEN}Wynik:${NC} $input (bin) = ${GREEN}$decimal${NC} (dec)"
            
            # Dla krótkich liczb pokaż proces
            if [ ${#input} -le 8 ]; then
                show_conversion_steps "$input"
            fi
        else
            echo -e "${RED}Błąd:${NC} '$input' nie jest liczbą binarną!"
            echo "Wpisz 'h' aby zobaczyć pomoc"
            echo ""
        fi
    done
fi

Adres IPv4

Format: XXX.XXX.XXX.XXX gdzie XXX to liczba 0-255

Prosty regex (akceptuje nieprawidłowe adresy jak 999.999.999.999):

[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}

Dokładniejszy regex:

((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)

Wyjaśnienie: - 25[0-5] - 250-255 - 2[0-4][0-9] - 200-249 - [01]?[0-9][0-9]? - 0-199 - \. - dosłowna kropka - {3} - trzy pierwsze oktety - Ostatni oktet bez kropki na końcu

Projekt: Szyfrowanie adresów IP

Analiza problemu: - Znaleźć wszystkie adresy IPv4 w pliku - Zaszyfrować każdy adres - Użyć szyfru zamiennego (np. ROT5 dla cyfr lub zamiana cyfr)

Szczegółowe rozwiązanie:

#!/bin/bash

# Domyślny plik
DEFAULT_FILE="test.txt"
file="${1:-$DEFAULT_FILE}"

# Kolory
RED='\033[0;31m'
GREEN='\033[0;32m'
BLUE='\033[0;34m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Sprawdzenie czy plik istnieje
if [ ! -f "$file" ]; then
    echo -e "${RED}Błąd: Plik '$file' nie istnieje!${NC}"
    echo "Użycie: $0 [nazwa_pliku]"
    exit 1
fi

# Tworzenie kopii zapasowej
timestamp=$(date +%Y%m%d_%H%M%S)
backup_file="${file}.backup_${timestamp}"
cp "$file" "$backup_file"
echo -e "${GREEN}Utworzono kopię zapasową:${NC} $backup_file"
echo ""

# Funkcja szyfrująca cyfry metodą ROT5
# 0→5, 1→6, 2→7, 3→8, 4→9, 5→0, 6→1, 7→2, 8→3, 9→4
encrypt_digit_rot5() {
    local digit=$1
    echo $(( (digit + 5) % 10 ))
}

# Funkcja szyfrująca cyfry metodą odwrotną
# 0→9, 1→8, 2→7, ..., 9→0
encrypt_digit_reverse() {
    local digit=$1
    echo $((9 - digit))
}

# Funkcja szyfrująca cyfry metodą XOR (z kluczem 7)
encrypt_digit_xor() {
    local digit=$1
    echo $((digit ^ 7))
}

# Wybór metody szyfrowania
ENCRYPT_METHOD="rot5"  # Opcje: rot5, reverse, xor

# Funkcja szyfrująca adres IP
encrypt_ip() {
    local ip=$1
    local encrypted=""
    
    for (( i=0; i<${#ip}; i++ )); do
        char="${ip:$i:1}"
        
        if [[ "$char" =~ [0-9] ]]; then
            # Szyfrowanie cyfry
            case $ENCRYPT_METHOD in
                rot5)
                    encrypted+=$(encrypt_digit_rot5 $char)
                    ;;
                reverse)
                    encrypted+=$(encrypt_digit_reverse $char)
                    ;;
                xor)
                    encrypted+=$(encrypt_digit_xor $char)
                    ;;
            esac
        else
            # Pozostałe znaki (kropka) bez zmian
            encrypted+="$char"
        fi
    done
    
    echo "$encrypted"
}

# Regex dla IPv4 (uproszczony)
ip_regex='([0-9]{1,3}\.){3}[0-9]{1,3}'

# Przetwarzanie pliku
temp_file="${file}.tmp"
> "$temp_file"

line_number=0
encrypted_count=0

echo -e "${BLUE}Przetwarzanie pliku...${NC}"
echo ""

while IFS= read -r line; do
    ((line_number++))
    
    # Sprawdzenie czy linia zawiera adres IP
    if [[ $line =~ $ip_regex ]]; then
        original_line="$line"
        new_line="$line"
        
        # Znajdowanie i szyfrowanie wszystkich adresów IP w linii
        while [[ $new_line =~ $ip_regex ]]; do
            ip="${BASH_REMATCH[0]}"
            encrypted=$(encrypt_ip "$ip")
            new_line="${new_line/$ip/$encrypted}"
            ((encrypted_count++))
        done
        
        # Wyświetlenie informacji o zmianie
        echo -e "${YELLOW}Linia $line_number:${NC}"
        echo -e "  Przed: $original_line"
        echo -e "  Po:    ${GREEN}$new_line${NC}"
        echo ""
        
        echo "$new_line" >> "$temp_file"
    else
        # Linia bez adresu IP - przepisz bez zmian
        echo "$line" >> "$temp_file"
    fi
done < "$file"

# Zastąpienie oryginalnego pliku
mv "$temp_file" "$file"

echo -e "${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo -e "${GREEN}Szyfrowanie zakończone!${NC}"
echo -e "${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo "Metoda szyfrowania: $ENCRYPT_METHOD"
echo "Zaszyfrowano adresów IP: $encrypted_count"
echo "Przetworzono linii: $line_number"
echo "Kopia zapasowa: $backup_file"
echo ""

# Opcja deszyfrowania
echo -e "${YELLOW}Aby odszyfrować plik, uruchom skrypt na zaszyfrowanym pliku ponownie${NC}"
echo -e "${YELLOW}(ROT5 jest samoodwrotny: szyfrowanie = deszyfrowanie)${NC}"

Szczegóły implementacji:

  1. BASH_REMATCH:

    ip="${BASH_REMATCH[0]}"

CSV (Comma-Separated Values) to format pliku tekstowego do przechowywania danych tabelarycznych.

Cechy: - Wartości oddzielone separatorem (przecinek, średnik, tabulator) - Każda linia to jeden rekord - Pierwsza linia często zawiera nagłówki

Projekt: Sortowanie według wieku

Analiza problemu: - Plik CSV z formatem: Imię_I_Nazwisko;Rok_urodzenia;Miesiąc_urodzenia - Obliczyć wiek każdej osoby - Posortować według wieku - Utworzyć pliki z 10 najmłodszymi i 10 najstarszymi

Szczegółowe rozwiązanie:

#!/bin/bash

# Domyślny plik wejściowy
INPUT_FILE="${1:-dane.csv}"

# Kolory
RED='\033[0;31m'
GREEN='\033[0;32m'
BLUE='\033[0;34m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Sprawdzenie czy plik istnieje
if [ ! -f "$INPUT_FILE" ]; then
    echo -e "${RED}Błąd: Plik '$INPUT_FILE' nie istnieje!${NC}"
    echo "Użycie: $0 [plik_csv]"
    exit 1
fi

# Obecny rok i miesiąc
CURRENT_YEAR=$(date +%Y)
CURRENT_MONTH=$(date +%m)

echo -e "${BLUE}╔═══════════════════════════════════════════╗${NC}"
echo -e "${BLUE}║  PRZETWARZANIE DANYCH CSV - SORTOWANIE    ║${NC}"
echo -e "${BLUE}╚═══════════════════════════════════════════╝${NC}"
echo ""
echo "Plik wejściowy: $INPUT_FILE"
echo "Obecna data: $CURRENT_MONTH/$CURRENT_YEAR"
echo ""

# Pobieranie nagłówka
header=$(head -n 1 "$INPUT_FILE")
echo -e "${YELLOW}Nagłówek:${NC} $header"
echo ""

# Sprawdzenie formatu nagłówka
if [[ ! "$header" =~ Imię.*Nazwisko.*Rok.*Miesiąc ]]; then
    echo -e "${YELLOW}Ostrzeżenie: Nagłówek nie pasuje do oczekiwanego formatu${NC}"
    echo "Oczekiwany format: Imię_I_Nazwisko;Rok_urodzenia;Miesiąc_urodzenia"
    echo ""
fi

# Funkcja obliczająca wiek
calculate_age() {
    local birth_year=$1
    local birth_month=$2
    local age=$((CURRENT_YEAR - birth_year))
    
    # Jeśli urodziny jeszcze nie było w tym roku, odejmij 1
    if [ $birth_month -gt $CURRENT_MONTH ]; then
        age=$((age - 1))
    fi
    
    echo $age
}

# Tworzenie pliku tymczasowego z obliczonym wiekiem
temp_file="temp_with_age.csv"
> "$temp_file"

echo -e "${BLUE}Obliczanie wieku dla każdej osoby...${NC}"
echo ""

line_count=0

# Przetwarzanie danych (pomijamy nagłówek)
tail -n +2 "$INPUT_FILE" | while IFS=';' read -r name year month; do
    # Walidacja danych
    if [[ ! "$year" =~ ^[0-9]+$ ]] || [[ ! "$month" =~ ^[0-9]+$ ]]; then
        echo -e "${YELLOW}Pomijam nieprawidłowy rekord: $name;$year;$month${NC}"
        continue
    fi
    
    # Walidacja roku
    if [ $year -lt 1900 ] || [ $year -gt $CURRENT_YEAR ]; then
        echo -e "${YELLOW}Pomijam rekord z nieprawidłowym rokiem: $name;$year;$month${NC}"
        continue
    fi
    
    # Walidacja miesiąca
    if [ $month -lt 1 ] || [ $month -gt 12 ]; then
        echo -e "${YELLOW}Pomijam rekord z nieprawidłowym miesiącem: $name;$year;$month${NC}"
        continue
    fi
    
    # Obliczanie wieku
    age=$(calculate_age $year $month)
    
    # Zapisywanie z wiekiem na początku (do sortowania)
    # Format: wiek;nazwa;rok;miesiąc
    echo "$age;$name;$year;$month" >> "$temp_file"
    
    ((line_count++))
done

echo -e "${GREEN}Przetworzono $line_count rekordów${NC}"
echo ""

# Sprawdzenie czy są dane do przetworzenia
if [ ! -s "$temp_file" ] || [ $line_count -eq 0 ]; then
    echo -e "${RED}Błąd: Brak prawidłowych danych do przetworzenia${NC}"
    rm -f "$temp_file"
    exit 1
fi

# Sortowanie według wieku (kolumna 1, numerycznie)

Projekt 1: Backup System

Automatyczny system tworzący kopie zapasowe z rotacją:

#!/bin/bash

SOURCE_DIR="$1"
BACKUP_DIR="${2:-./backups}"
MAX_BACKUPS=7

if [ -z "$SOURCE_DIR" ]; then
    echo "Użycie: $0 <katalog_źródłowy> [katalog_backupów]"
    exit 1
fi

# Tworzenie katalogu backupów
mkdir -p "$BACKUP_DIR"

# Tworzenie timestampu
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_NAME="backup_${TIMESTAMP}.tar.gz"

# Tworzenie archiwum
tar -czf "${BACKUP_DIR}/${BACKUP_NAME}" -C "$(dirname "$SOURCE_DIR")" "$(basename "$SOURCE_DIR")"

echo "Utworzono backup: ${BACKUP_NAME}"

# Rotacja - usuwanie starych backupów
cd "$BACKUP_DIR" || exit
ls -t backup_*.tar.gz | tail -n +$((MAX_BACKUPS + 1)) | xargs -r rm
echo "Zachowano ostatnie $MAX_BACKUPS backupów"

Projekt 2: Log Analyzer

Analiza plików logów:

#!/bin/bash

LOG_FILE="$1"

if [ ! -f "$LOG_FILE" ]; then
    echo "Błąd: Plik nie istnieje"
    exit 1
fi

echo "=== ANALIZA LOGU: $LOG_FILE ==="
echo ""

# Liczba linii
total_lines=$(wc -l < "$LOG_FILE")
echo "Całkowita liczba linii: $total_lines"

# Liczba błędów
errors=$(grep -ci "error" "$LOG_FILE")
echo "Błędy (ERROR): $errors"

# Liczba ostrzeżeń
warnings=$(grep -ci "warning" "$LOG_FILE")
echo "Ostrzeżenia (WARNING): $warnings"

# Najczęstsze komunikaty
echo ""
echo "Top 5 najczęstszych komunikatów:"
grep -oE "ERROR:.*|WARNING:.*" "$LOG_FILE" | sort | uniq -c | sort -rn | head -5

# Statystyki czasowe
echo ""
echo "Pierwsze zdarzenie: $(head -1 "$LOG_FILE" | cut -d' ' -f1-2)"
echo "Ostatnie zdarzenie: $(tail -1 "$LOG_FILE" | cut -d' ' -f1-2)"

Projekt 3: System Monitor

Monitor zasobów systemowych:

#!/bin/bash

while true; do
    clear
    echo "=== MONITOR SYSTEMU ==="
    echo "Czas: $(date '+%Y-%m-%d %H:%M:%S')"
    echo ""
    
    # CPU
    echo "CPU Usage:"
    top -bn1 | grep "Cpu(s)" | sed "s/.*, *\([0-9.]*\)%* id.*/\1/" | awk '{print "  "100 - $1"%"}'
    
    # Pamięć
    echo ""
    echo "Pamięć:"
    free -h | grep Mem | awk '{print "  Użyta: "$3" / "$2" ("int($3/$2*100)"%)"}'
    
    # Dysk
    echo ""
    echo "Dysk:"
    df -h / | tail -1 | awk '{print "  Użyte: "$3" / "$2" ("$5")"}'
    
    # Top procesy
    echo ""
    echo "Top 5 procesów (CPU):"
    ps aux --sort=-%cpu | head -6 | tail -5 | awk '{printf "  %-20s %5s%%\n", $11, $3}'
    
    sleep 5
done
