Gotowe wzorce kodu - Bash

Przykłady gotowe do skopiowania i użycia na egzaminie.

1. Walidacja argumentów

Sprawdzenie liczby argumentów

#!/bin/bash

# Sprawdź czy podano dokładnie 2 argumenty
if [ $# -ne 2 ]; then
    echo "Błąd: Wymagane 2 argumenty"
    echo "Użycie: $0 <argument1> <argument2>"
    exit 1
fi

# Użycie argumentów
plik=$1
fraza=$2
echo "Plik: $plik, Fraza: $fraza"

Sprawdzenie minimalnej liczby argumentów

#!/bin/bash

if [ $# -lt 1 ]; then
    echo "Błąd: Potrzebny co najmniej 1 argument"
    exit 1
fi

2. Sprawdzanie plików

Sprawdź czy plik istnieje

#!/bin/bash

plik=$1

if [ ! -f "$plik" ]; then
    echo "Błąd: Plik '$plik' nie istnieje!"
    exit 1
fi

echo "Plik istnieje, przetwarzam..."

Sprawdź czy katalog istnieje

#!/bin/bash

katalog=$1

if [ ! -d "$katalog" ]; then
    echo "Katalog nie istnieje, tworzę..."
    mkdir -p "$katalog"
fi

cd "$katalog"

Sprawdź uprawnienia

#!/bin/bash

plik=$1

if [ ! -r "$plik" ]; then
    echo "Błąd: Brak uprawnień do odczytu"
    exit 1
fi

if [ ! -w "$plik" ]; then
    echo "Błąd: Brak uprawnień do zapisu"
    exit 1
fi

3. Pętle

Pętla po wszystkich argumentach

#!/bin/bash

# Iteracja po wszystkich argumentach
for arg in "$@"; do
    echo "Argument: $arg"
done

Pętla po liczbach

#!/bin/bash

# Od 1 do 10
for i in {1..10}; do
    echo "Liczba: $i"
done

# Składnia C-style
for (( i=0; i<10; i++ )); do
    echo "i = $i"
done

Pętla po plikach

#!/bin/bash

# Wszystkie pliki .txt w bieżącym katalogu
for plik in *.txt; do
    echo "Przetwarzam: $plik"
done

# Rekurencyjnie (wymaga shopt -s globstar)
shopt -s globstar
for plik in **/*.txt; do
    echo "Znaleziono: $plik"
done

Czytanie pliku linia po linii

# Pomiń pierwszą linię (nagłówek)
tail -n +2 "$plik" | while IFS= read -r linia; do

Zliczanie linii

#!/bin/bash

5. Przetwarzanie CSV

Czytanie CSV z separatorem średnika

#!/bin/bash

plik="dane.csv"

while IFS=';' read -r kolumna1 kolumna2 kolumna3; do
    echo "Kol1: $kolumna1"
    echo "Kol2: $kolumna2"
    echo "Kol3: $kolumna3"
    echo "---"
done < "$plik"

CSV z pominięciem nagłówka

#!/bin/bash

plik="dane.csv"

# Pobierz nagłówek
naglowek=$(head -1 "$plik")
echo "Nagłówek: $naglowek"

# Przetwórz dane (bez nagłówka)
tail -n +2 "$plik" | while IFS=';' read -r imie rok miesiac; do
    echo "Imię: $imie, Rok: $rok, Miesiąc: $miesiac"
done

Sortowanie CSV numerycznie

#!/bin/bash

# Sortuj według 2. kolumny numerycznie
sort -t';' -k2n dane.csv > posortowane.csv

# Sortuj malejąco (od największego)
sort -t';' -k2nr dane.csv > posortowane.csv

6. Tablice i losowanie

Deklaracja i użycie tablicy

#!/bin/bash

# Deklaracja tablicy
opcje=("Kamień" "Papier" "Nożyce")

# Dostęp do elementów
echo ${opcje[0]}        # Kamień
echo ${opcje[1]}        # Papier
echo ${opcje[2]}        # Nożyce

# Wszystkie elementy
echo ${opcje[@]}

# Długość tablicy
echo ${#opcje[@]}

Losowanie z tablicy

#!/bin/bash

choices=("K" "P" "N")

# Losowy indeks (0-2)
indeks=$((RANDOM % 3))

# Wylosowany element
wybor=${choices[$indeks]}
echo "Wylosowano: $wybor"

Losowanie liczby z zakresu

#!/bin/bash

# Liczba od 0 do 9
liczba=$((RANDOM % 10))

# Liczba od 1 do 10
liczba=$((RANDOM % 10 + 1))

# Liczba z zakresu min-max
min=5
max=15
liczba=$((RANDOM % (max - min + 1) + min))

7. Konwersje i transformacje

Zamiana małych liter na wielkie

#!/bin/bash

tekst="hello world"

# Metoda 1: tr
wielkie=$(echo "$tekst" | tr a-z A-Z)
echo $wielkie    # HELLO WORLD

# Metoda 2: bash 4.0+
wielkie=${tekst^^}
echo $wielkie    # HELLO WORLD

Konwersja systemów liczbowych

#!/bin/bash

# Binarny na dziesiętny
binary="1010"
decimal=$((2#$binary))
echo "$binary (bin) = $decimal (dec)"    # 10

# Szesnastkowy na dziesiętny
hex="FF"
decimal=$((16#$hex))
echo "$hex (hex) = $decimal (dec)"       # 255

# Ósemkowy na dziesiętny
octal="77"
decimal=$((8#$octal))
echo "$octal (oct) = $decimal (dec)"     # 63

Zamiana znaków (tr)

#!/bin/bash

# ROT13 dla liter
echo "hello" | tr 'a-zA-Z' 'n-za-mN-ZA-M'    # uryyb

# ROT5 dla cyfr (szyfr zamienny)
echo "192.168.1.1" | tr '0123456789' '5678901234'    # 647.613.6.6

# Usunięcie znaków
echo "a-b-c" | tr -d '-'    # abc

8. Wyrażenia regularne

Walidacja liczby binarnej

#!/bin/bash

liczba=$1

if [[ $liczba =~ ^[01]+$ ]]; then
    echo "To jest liczba binarna"
    decimal=$((2#$liczba))
    echo "Wartość dziesiętna: $decimal"
else
    echo "Błąd: To nie jest liczba binarna!"
    exit 1
fi

Sprawdzanie adresu IPv4 (uproszczone)

#!/bin/bash

ip=$1
regex='^([0-9]{1,3}\.){3}[0-9]{1,3}$'

if [[ $ip =~ $regex ]]; then
    echo "To wygląda jak adres IP"
else
    echo "To nie jest adres IP"
fi

Wyszukiwanie IPv4 w tekście

#!/bin/bash

plik="test.txt"
ip_regex='([0-9]{1,3}\.){3}[0-9]{1,3}'

while IFS= read -r linia; do
    if [[ $linia =~ $ip_regex ]]; then
        echo "Znaleziono IP: ${BASH_REMATCH[0]}"
        echo "W linii: $linia"
    fi
done < "$plik"

9. Funkcje

Funkcja z argumentami

#!/bin/bash

powitanie() {
    local imie=$1
    local wiek=$2
    echo "Witaj $imie, masz $wiek lat"
}

# Wywołanie
powitanie "Jan" 25

Funkcja zwracająca wartość przez echo

#!/bin/bash

oblicz_wiek() {
    local rok_urodzenia=$1
    local obecny_rok=2026
    local wiek=$((obecny_rok - rok_urodzenia))
    echo $wiek
}

# Wywołanie i przechwycenie wyniku
wiek=$(oblicz_wiek 1990)
echo "Wiek: $wiek"

Funkcja z kodem powrotu

#!/bin/bash

czy_parzysta() {
    local liczba=$1
    if [ $((liczba % 2)) -eq 0 ]; then
        return 0    # sukces (parzysta)
    else
        return 1    # błąd (nieparzysta)
    fi
}

# Wywołanie
if czy_parzysta 4; then
    echo "Liczba jest parzysta"
else
    echo "Liczba jest nieparzysta"
fi

10. Operacje na stringach

Długość stringa

#!/bin/bash

tekst="Hello World"
dlugosc=${#tekst}
echo "Długość: $dlugosc"    # 11

Substring (wycinek)

#!/bin/bash

tekst="Hello World"

# Od pozycji 0, długość 5
fragment=${tekst:0:5}
echo $fragment    # Hello

# Od pozycji 6 do końca
fragment=${tekst:6}
echo $fragment    # World

Zamiana w stringu

#!/bin/bash

tekst="jabłko gruszka jabłko"

# Zamień pierwsze wystąpienie
nowy=${tekst/jabłko/pomarańcza}
echo $nowy    # pomarańcza gruszka jabłko

# Zamień wszystkie wystąpienia
nowy=${tekst//jabłko/pomarańcza}
echo $nowy    # pomarańcza gruszka pomarańcza

Usuwanie prefiksu/sufiksu

#!/bin/bash

plik="dokument.txt.backup"

# Usuń najmniejszy pasujący sufiks
bez_backup=${plik%.backup}
echo $bez_backup    # dokument.txt

# Usuń rozszerzenie
bez_rozszerzenia=${plik%.*}
echo $bez_rozszerzenia    # dokument.txt

# Usuń prefix
plik="/home/user/dokument.txt"
tylko_nazwa=${plik##*/}
echo $tylko_nazwa    # dokument.txt

11. Przykłady kompletnych zadań

Szybka szachownica (zwięzła wersja)

#!/bin/bash

mkdir -p szachownica && cd szachownica

for col in {a..h}; do
    for row in {1..8}; do
        mkdir -p "${col}${row}"
        col_num=$(($(printf "%d" "'$col") - 96))
        [ $(( (col_num + row) % 2 )) -eq 0 ] && color="bialy" || color="czarny"
        
        case "$row" in
            2) piece="WP" ;;
            7) piece="BP" ;;
            1) case $col in a|h) piece="WR";; b|g) piece="WN";; c|f) piece="WB";; d) piece="WQ";; e) piece="WK";; esac ;;
            8) case $col in a|h) piece="BR";; b|g) piece="BN";; c|f) piece="BB";; d) piece="BQ";; e) piece="BK";; esac ;;
            *) piece="" ;;
        esac
        
        echo "$piece" > "${col}${row}/$color"
    done
done

#!/bin/bash

choices=(K P N)
read -p "Wybierz K/P/N: " p
p=$(echo "$p" | tr a-z A-Z)

echo "Ty: $p, Komputer: $c"
