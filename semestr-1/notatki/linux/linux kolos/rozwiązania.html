Rozwiązania zadań - Bash

Zadanie 1: Szachownica

Skrypt tworzy katalog szachownica z podfolderami dla każdego pola (a1-h8), w każdym tworzy plik z kolorem pola, a dla pól startowych dodaje nazwy figur.

#!/bin/bash
mkdir -p szachownica && cd szachownica  # Tworzy folder i wchodzi do niego

for col in {a..h}; do                    # Pętla po kolumnach a-h
    for row in {1..8}; do                # Pętla po wierszach 1-8
        mkdir -p "${col}${row}"          # Tworzy folder pola np. "a1"
        col_num=$(($(printf "%d" "'$col") - 96))  # Konwertuje literę na liczbę (a=1, b=2...)
        [ $(( (col_num + row) % 2 )) -eq 0 ] && color="bialy" || color="czarny"  # Parzysta suma = biały
        
        case "$row" in
            2) piece="WP" ;;             # Wiersz 2: białe piony (White Pawn)
            7) piece="BP" ;;             # Wiersz 7: czarne piony (Black Pawn)
            1) case $col in a|h) piece="WR";; b|g) piece="WN";; c|f) piece="WB";; d) piece="WQ";; e) piece="WK";; esac ;;  # Białe figury
            8) case $col in a|h) piece="BR";; b|g) piece="BN";; c|f) piece="BB";; d) piece="BQ";; e) piece="BK";; esac ;;  # Czarne figury
            *) piece="" ;;               # Puste pola (wiersze 3-6)
        esac
        
        echo "$piece" > "${col}${row}/$color"  # Zapisuje figurę do pliku o nazwie koloru
    done
done

Zadanie 6: Przetwarzanie CSV - sortowanie według wieku

Skrypt przetwarza plik CSV, sortuje według wieku i tworzy pliki z 10 najmłodszymi i najstarszymi osobami.

#!/bin/bash
header=$(head -1 dane.csv)               # Pobierz nagłówek (pierwsza linia)
# tail -n +2 = pomiń nagłówek, awk dodaje wiek na początek, sort sortuje numerycznie
sorted=$(tail -n +2 dane.csv | awk -F';' '{print 2026-$2";"$0}' | sort -t';' -k1n)

echo "$header" > new.csv                 # Nagłówek do pliku młodszych
echo "$sorted" | head -10 | cut -d';' -f2- >> new.csv  # 10 najmłodszych (bez kolumny wieku)

echo "$header" > old.csv                 # Nagłówek do pliku starszych
echo "$sorted" | tail -10 | cut -d';' -f2- >> old.csv  # 10 najstarszych

Uwagi

  • Skrypty są maksymalnie uproszczone - bez walidacji danych wejściowych
  • Aby uruchomić skrypt, zapisz go do pliku, nadaj uprawnienia wykonywania (chmod +x skrypt.sh) i uruchom (./skrypt.sh)

Zadanie 1: Szachownica (szczegółowo)

#!/bin/bash
# ============================================
# ZADANIE: Stwórz strukturę folderów szachownicy
# ============================================

# Tworzy główny katalog "szachownica" (-p = nie zgłasza błędu jeśli istnieje)
# && oznacza: wykonaj następne polecenie tylko jeśli poprzednie się powiodło
mkdir -p szachownica && cd szachownica

# Zewnętrzna pętla: iteruje po kolumnach od 'a' do 'h'
# {a..h} to rozwinięcie bash - generuje: a b c d e f g h
for col in {a..h}; do

    # Wewnętrzna pętla: iteruje po wierszach od 1 do 8
    for row in {1..8}; do
    
        # Tworzy folder dla pola, np. "a1", "b2", itd.
        mkdir -p "${col}${row}"
        
        # KONWERSJA LITERY NA LICZBĘ:
        # printf "%d" "'a" zwraca kod ASCII litery 'a' (97)
        # Odejmujemy 96, więc: a=1, b=2, c=3, ..., h=8
        col_num=$(($(printf "%d" "'$col") - 96))
        
        # OBLICZANIE KOLORU POLA:
        # Na szachownicy pole jest BIAŁE gdy suma kolumny i wiersza jest PARZYSTA
        # Przykład: a1 -> 1+1=2 (parzyste) -> ale a1 jest CZARNE!
        # Więc: parzysta suma = biały, nieparzysta = czarny
        # [ warunek ] && akcja1 || akcja2  = if-then-else w jednej linii
        [ $(( (col_num + row) % 2 )) -eq 0 ] && color="bialy" || color="czarny"
        
        # PRZYPISANIE FIGURY na podstawie wiersza:
        # case to odpowiednik switch z innych języków
        case "$row" in
            2) piece="WP" ;;   # Wiersz 2: białe piony (W=White, P=Pawn)
            7) piece="BP" ;;   # Wiersz 7: czarne piony (B=Black, P=Pawn)
            
            # Wiersz 1: białe figury - zagnieżdżony case dla kolumn
            1) case $col in 
                a|h) piece="WR";;  # a1, h1: wieże (R=Rook)
                b|g) piece="WN";;  # b1, g1: skoczki (N=Knight)
                c|f) piece="WB";;  # c1, f1: gońce (B=Bishop)
                d) piece="WQ";;    # d1: hetman (Q=Queen)
                e) piece="WK";;    # e1: król (K=King)
               esac ;;
            
            # Wiersz 8: czarne figury (analogicznie)
            8) case $col in 
                a|h) piece="BR";;  # Czarne wieże
                b|g) piece="BN";;  # Czarne skoczki
                c|f) piece="BB";;  # Czarni gońce
                d) piece="BQ";;    # Czarny hetman
                e) piece="BK";;    # Czarny król
               esac ;;
            
            # Wiersze 3-6: puste pola
            *) piece="" ;;
        esac
        
        # Zapisuje figurę (lub pusty string) do pliku
        # Nazwa pliku = kolor pola (bialy lub czarny)
        echo "$piece" > "${col}${row}/$color"
        
    done  # Koniec pętli wierszy
done      # Koniec pętli kolumn

# > tmp     = zapisz wynik do pliku "tmp" (nadpisz jeśli istnieje)
# &&        = wykonaj następne polecenie tylko jeśli poprzednie się powiodło
# mv tmp "$1" = przenieś (zmień nazwę) tmp na oryginalną nazwę pliku

grep -v "$2" "$1" > tmp && mv tmp "$1"

# ALTERNATYWNIE można użyć sed:
# sed -i "/$2/d" "$1"
# -i = edytuj plik w miejscu
# /$2/d = usuń linie pasujące do wzorca $2

Zadanie 3: Gra w kamień, papier, nożyce (szczegółowo)

#!/bin/bash
# ============================================
# ZADANIE: Gra kamień, papier, nożyce
# ============================================

# TABLICA w bashu:
# Indeksy: 0=K, 1=P, 2=N
choices=(K P N)

# READ wczytuje dane od użytkownika:
# -p "tekst" = wyświetl prompt przed wczytaniem
# Wartość zapisywana do zmiennej p
read -p "Wybierz K/P/N: " p

# ZAMIANA NA WIELKIE LITERY:
# echo "$p" wypisuje wartość zmiennej
# | (pipe) przekazuje wynik do następnego polecenia
# tr a-z A-Z zamienia małe litery na wielkie
# $(...) przechwytuje wynik polecenia do zmiennej
p=$(echo "$p" | tr a-z A-Z)

# LOSOWANIE WYBORU KOMPUTERA:
# $RANDOM = wbudowana zmienna bash, losowa liczba 0-32767
# % 3 = reszta z dzielenia przez 3 (wynik: 0, 1 lub 2)
# ${choices[indeks]} = element tablicy o danym indeksie
c=${choices[$((RANDOM % 3))]}

# Wyświetl wybory obu graczy
echo "Ty: $p, Komputer: $c"

# SPRAWDZENIE REMISU:
# [ "$p" == "$c" ] = porównanie stringów
# && = jeśli prawda, wykonaj następne
# exit = zakończ skrypt
[ "$p" == "$c" ] && echo "REMIS!" && exit

# SPRAWDZENIE WYGRANEJ:
# [[ ... =~ wzorzec ]] = dopasowanie do wyrażenia regularnego
# "$p$c" łączy oba wybory, np. "KN" (gracz K, komputer N)
# ^(KN|PK|NP)$ = wzorzec wygrywających kombinacji:
#   KN = Kamień bije Nożyce
#   PK = Papier bije Kamień  
#   NP = Nożyce biją Papier
# ^ = początek stringa, $ = koniec stringa
# | = lub (alternatywa)
[[ "$p$c" =~ ^(KN|PK|NP)$ ]] && echo "WYGRANA!" || echo "PRZEGRANA!"

Zadanie 4: Konwerter binarny na dziesiętny (szczegółowo)

#!/bin/bash
# ============================================
# ZADANIE: Konwertuj liczby binarne na dziesiętne
# Użycie: ./skrypt.sh 1010 1111 10000
# ============================================

# ZMIENNE SPECJALNE ARGUMENTÓW:
# $@ = wszystkie argumenty jako osobne słowa
# $* = wszystkie argumenty jako jeden string
# $# = liczba argumentów
# $0 = nazwa skryptu
# $1, $2, ... = konkretne argumenty

# PĘTLA FOR po wszystkich argumentach:
# "$@" w cudzysłowach zachowuje argumenty ze spacjami
for b in "$@"; do

    # KONWERSJA SYSTEMÓW LICZBOWYCH W BASH:
    # $((wyrażenie)) = obliczenia arytmetyczne
    # BAZA#LICZBA = konwersja z systemu o podstawie BAZA
    #
    # Przykłady:
    # $((2#1010))  = 10  (binarny -> dziesiętny)
    # $((16#FF))   = 255 (szesnastkowy -> dziesiętny)
    # $((8#77))    = 63  (ósemkowy -> dziesiętny)
    #
    # Jak działa 2#1010:
    # 1*2³ + 0*2² + 1*2¹ + 0*2⁰ = 8 + 0 + 2 + 0 = 10
    
    echo "$b (bin) = $((2#$b)) (dec)"
    
done

# PRZYKŁADOWE WYWOŁANIE:
# ./skrypt.sh 1010 1111 10000
# Wynik:
# 1010 (bin) = 10 (dec)
# 1111 (bin) = 15 (dec)
# 10000 (bin) = 16 (dec)

Zadanie 5: Szyfrowanie adresów IP (szczegółowo)

#!/bin/bash
# ============================================
# ZADANIE: Zaszyfruj cyfry w pliku szyfrem ROT5
# ============================================

# SZYFR ROT5 DLA CYFR:
# Każda cyfra jest zamieniana na cyfrę przesuniętą o 5 pozycji
# 0 -> 5
# 1 -> 6
# 2 -> 7
# 3 -> 8
# 4 -> 9
# 5 -> 0  (zawijamy do początku)
# 6 -> 1
# 7 -> 2
# 8 -> 3
# 9 -> 4

# POLECENIE TR (translate):
# tr 'zestaw1' 'zestaw2' - zamienia znaki z zestawu1 na odpowiadające z zestawu2
# Znaki są mapowane 1:1 na podstawie pozycji:
#   0123456789
#   5678901234
# Czyli: 0->5, 1->6, 2->7, itd.

# PRZEKIEROWANIA:
# < test.txt  = wczytaj dane z pliku (stdin)
# > tmp       = zapisz wynik do pliku tmp
# && mv ...   = jeśli sukces, przenieś tmp na miejsce oryginału

tr '0123456789' '5678901234' < test.txt > tmp && mv tmp test.txt

# PRZYKŁAD:
# Przed: 192.168.1.1
# Po:    647.613.6.6
#
# Szyfr jest symetryczny - zastosowany dwukrotnie przywraca oryginał!
# ROT5(ROT5(x)) = x

Zadanie 6: Przetwarzanie CSV - sortowanie według wieku (szczegółowo)

#!/bin/bash
# ============================================
# ZADANIE: Sortuj CSV według wieku, utwórz pliki
# z 10 najmłodszymi i 10 najstarszymi osobami
# ============================================

# Format pliku dane.csv:
# Imię_I_nazwisko;Rok_urodzenia;Miesiąc_urodzenia
# Jan Kowalski;1990;5
# Anna Nowak;1985;12
# ...

# KROK 1: Pobierz nagłówek (pierwszą linię)
# head -1 = wyświetl tylko pierwszą linię
# head -n 1 = to samo (bardziej czytelne)
header=$(head -1 dane.csv)

# KROK 2: Przetwórz dane (bez nagłówka) i posortuj
# 
# tail -n +2     = wyświetl od 2. linii do końca (pomiń nagłówek)
#                  +2 oznacza "zaczynając od linii 2"
#
# | (pipe)       = przekaż wynik do następnego polecenia
#
# awk -F';'      = użyj średnika jako separatora pól
#   '{print 2026-$2";"$0}'
#   $2 = drugie pole (rok urodzenia)
#   $0 = cała linia
#   2026-$2 = oblicz wiek
#   Wynik: "wiek;oryginalna_linia"
#   Np: "36;Jan Kowalski;1990;5"
#
# sort -t';' -k1n
#   -t';'  = separator pól to średnik
#   -k1    = sortuj według pierwszego pola (wiek)
#   n      = sortuj numerycznie (nie alfabetycznie!)
#            Bez 'n': 9 > 10 (bo "9" > "1" alfabetycznie)
#            Z 'n':   9 < 10 (poprawne porównanie liczbowe)

sorted=$(tail -n +2 dane.csv | awk -F';' '{print 2026-$2";"$0}' | sort -t';' -k1n)

# KROK 3: Utwórz plik z 10 najmłodszymi
# > new.csv = utwórz/nadpisz plik, zapisz nagłówek
echo "$header" > new.csv

# head -10       = weź pierwsze 10 linii (najmłodsi mają najmniejszy wiek)
# cut -d';' -f2- = usuń pierwszą kolumnę (wiek), zostaw resztę
#   -d';'  = separator to średnik
#   -f2-   = pola od 2. do końca (pomijamy pole 1 czyli wiek)
# >> new.csv = dopisz do pliku (nie nadpisuj!)
echo "$sorted" | head -10 | cut -d';' -f2- >> new.csv

# KROK 4: Utwórz plik z 10 najstarszymi
echo "$header" > old.csv

# tail -10 = weź ostatnie 10 linii (najstarsi mają największy wiek)
echo "$sorted" | tail -10 | cut -d';' -f2- >> old.csv

# PRZYKŁAD DZIAŁANIA:
# 
# dane.csv:
# Imię;Rok;Miesiąc
# Jan;1990;5      -> wiek 36
# Anna;2000;12    -> wiek 26
# Piotr;1950;1    -> wiek 76
#
# Po przetworzeniu awk:
# 36;Jan;1990;5
# 26;Anna;2000;12
# 76;Piotr;1950;1
#
# Po sortowaniu (od najmłodszego):
# 26;Anna;2000;12
# 36;Jan;1990;5
# 76;Piotr;1950;1
#
# new.csv (najmłodsi, head):
# Imię;Rok;Miesiąc
# Anna;2000;12
# Jan;1990;5
#
# old.csv (najstarsi, tail):
# Imię;Rok;Miesiąc
# Piotr;1950;1
