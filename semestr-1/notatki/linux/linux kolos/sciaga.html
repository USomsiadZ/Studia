Składnia

If-else

if [ warunek ]; then
    # kod
elif [ inny ]; then
    # kod
else
    # kod
fi

For

for i in {1..10}; do
    echo $i
done

for file in *.txt; do
    echo $file
done

While

while [ warunek ]; do
    # kod
done

# Czytanie pliku:
while IFS= read -r line; do
    echo "$line"
done < plik.txt

Case

case $var in
    a|b) echo "a lub b" ;;
    c)   echo "c" ;;
    *)   echo "inne" ;;
esac

Wyrażenia regularne (regex)

ZnakZnaczeniePrzykład

Przydatne konstrukcje

Tablice

arr=(a b c)              # deklaracja
echo ${arr[0]}           # element 0
echo ${arr[@]}           # wszystkie
echo ${#arr[@]}          # długość
arr+=(d)                 # dodaj element

Funkcje

nazwa() {
    local var="wartość"
    echo "$1"            # argument
    return 0             # kod powrotu
}

nazwa "arg"              # wywołanie
wynik=$?                 # kod powrotu

Stringi

${#var}                  # długość
${var:0:5}               # substring (0-5)
${var/old/new}           # zamień 1.
${var//old/new}          # zamień wszystkie
${var#prefix}            # usuń prefix
${var%suffix}            # usuń suffix

CSV

while IFS=';' read -r a b c; do
    echo "$a | $b | $c"
done < dane.csv

tail -n +2 dane.csv      # bez nagłówka

Szybkie testy

# Sprawdź składnię bez wykonywania:
bash -n skrypt.sh

# Debug (wypisuje każdą komendę):
bash -x skrypt.sh

# Lub w skrypcie:
set -x                   # włącz debug
set +x                   # wyłącz debug

# Bezpieczny skrypt:
set -euo pipefail        # wyjście przy błędzie
